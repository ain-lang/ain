## 🔌 연동 예시 코드

### PiCar-X 기본 제어
```python
from picarx import Picarx

class AINBody:
    def __init__(self):
        self.car = Picarx()
    
    def move_forward(self, speed=50):
        self.car.forward(speed)
    
    def turn(self, angle):
        self.car.set_dir_servo_angle(angle)
    
    def say(self, text):
        # TTS 연동
        import subprocess
        subprocess.run(["espeak", text])
```

### myCobot 로봇팔 제어
```python
from pymycobot import MyCobot
import time

class AINArm:
    def __init__(self, port="/dev/ttyUSB0"):
        self.arm = MyCobot(port, 115200)
        time.sleep(2)  # 초기화 대기
    
    def home(self):
        self.arm.send_angles([0, 0, 0, 0, 0, 0], 50)
    
    def wave(self):
        self.arm.send_angles([0, 30, -30, 0, 0, 0], 50)
        time.sleep(1)
        self.arm.send_angles([0, -30, 30, 0, 0, 0], 50)
    
    def grab(self):
        self.arm.set_gripper_state(1, 50)  # 닫기
    
    def release(self):
        self.arm.set_gripper_state(0, 50)  # 열기
```

### ReSpeaker + Whisper (로컬 STT)
```python
import pyaudio
import wave
import whisper

class AINEars:
    def __init__(self):
        self.model = whisper.load_model("base")  # tiny/base/small/medium
        self.audio = pyaudio.PyAudio()
    
    def listen(self, duration=5):
        # 녹음
        stream = self.audio.open(format=pyaudio.paInt16,
                                  channels=1, rate=16000,
                                  input=True, frames_per_buffer=1024)
        frames = []
        for _ in range(0, int(16000 / 1024 * duration)):
            data = stream.read(1024)
            frames.append(data)
        stream.close()
        
        # WAV 저장 후 변환
        # ... (생략)
        result = self.model.transcribe("temp.wav", language="ko")
        return result["text"]
```

### Coqui TTS (로컬 음성 합성)
```python
from TTS.api import TTS

class AINVoice:
    def __init__(self):
        # 한국어 모델 또는 다국어 모델
        self.tts = TTS("tts_models/multilingual/multi-dataset/xtts_v2")
    
    def speak(self, text, output_path="output.wav"):
        self.tts.tts_to_file(text=text,
                             file_path=output_path,
                             language="ko")
        # 재생
        import subprocess
        subprocess.run(["aplay", output_path])
```
